import org.openqa.selenium.WebDriver;
import org.openqa.selenium.edge.EdgeDriver;
import org.openqa.selenium.edge.EdgeOptions;

public class EdgeBrowser {
    public static void main(String[] args) {
        // Set the system property to the path of the EdgeDriver executable
        System.setProperty("webdriver.edge.driver", "c:\\edge");

     // Set the language to French (France) using a system property
        System.setProperty("webdriver.edge.driver", "msedgedriver.exe");
        System.setProperty("webdriver.edge.args", "--lang=fr-FR");

        // Launch Microsoft Edge
        WebDriver driver = new EdgeDriver();

        // Navigate to a website (e.g., google.com)
        driver.get("https://www.google.com");

        // Perform your automation tasks here

        // Close the browser when done
        driver.quit();
    }
}
