import pyautogui
import time

def move_mouse_to_prevent_sleep(interval=60):
    try:
        while True:
            # Move the mouse to a different position
            pyautogui.moveRel(1, 1)
            pyautogui.moveRel(-1, -1)

            # Wait for the specified interval (in seconds)
            time.sleep(interval)

    except KeyboardInterrupt:
        print("Program terminated by user.")

if __name__ == "__main__":
    # You can adjust the interval (in seconds) as needed
    move_mouse_to_prevent_sleep(interval=60)
