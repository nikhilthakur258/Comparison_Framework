import org.openqa.selenium.WebDriver;
import org.openqa.selenium.edge.EdgeDriver;
import org.openqa.selenium.edge.EdgeOptions;
import java.util.Locale;

public class EdgeBrowserFrench {
    public static void main(String[] args) {
        // Set the system property to the path of the EdgeDriver executable
        System.setProperty("webdriver.edge.driver", "path_to_edge_driver_executable");

        // Set the EdgeOptions to configure the browser
        EdgeOptions options = new EdgeOptions();
        
        // Set the language to French (France)
        options.addArguments("--lang=fr-FR");

        // Launch Microsoft Edge with the configured options
        WebDriver driver = new EdgeDriver(options);

        // Navigate to a website (e.g., google.com)
        driver.get("https://www.google.com");

        // Perform your automation tasks here

        // Close the browser when done
        driver.quit();
    }
}
